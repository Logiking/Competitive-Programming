### STL unordered_set
___
````c++
#include <bits/stdc++.h>

using namespace std;

using i64 = long long;
using ull = unsigned long long;

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);

    int _;
    for (cin >> _; _; _ --) {
        int n;
        cin >> n;
    
        bool ok = false;
        unordered_set<string> ex;
        vector<string> s(n);
        for (auto &si : s) {
            cin >> si;
            if (ex.count(si))
                ok = true;
            ex.insert(si);
        }

        for (int i = 0; i < n; i ++) {
            for (int j = 1; j < int(s[i].size()); j ++)
                if (ex.count(s[i].substr(0, j))) {
                    ok = true;
                    break;
                }
            if (ok)
                break;
        }

        cout << (ok ? "NO\n" : "YES\n");

    }

    return 0;
}
````

````c++
#include <bits/stdc++.h>

using namespace std;

using i64 = long long;
using ull = unsigned long long;

const int N = 1e5 + 10;
int Trie[N][10], pos;
bool vis[N];

int insert(string &s) {
    int p = 0;
    bool pre_of_other = true, have_pre = false;
    for (int i = 0; i < int(s.size()); i ++) {
        if (!Trie[p][s[i] - '0']) {
            pre_of_other = false;
            Trie[p][s[i] - '0'] = ++ pos;
        }
        p = Trie[p][s[i] - '0'];
        if (vis[p])
            have_pre = true;
    }
    vis[p] = true;

    return !pre_of_other && !have_pre;
}

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);

    int _;
    for (cin >> _; _; _ --) {
        memset(Trie, 0, sizeof Trie);
        memset(vis, false, sizeof vis);
        pos = 0;

        int n;
        cin >> n;

        bool ok = true;
        while (n --) {
            string s;
            cin >> s;

            if (!insert(s))
                ok = false;
        }

        cout << (ok ? "YES" : "NO") << "\n";

    }

    return 0;
}
````
